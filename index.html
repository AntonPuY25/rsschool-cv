   export const getFilteringField = (dealsByDate: Deal[], filterField: string) =>
        dealsByDate.reduse(
            (acc: SelectedDateDeals, currentValue: Deal | CommonDeal) => {
                if (currentValue) {
                    const getDate = (value: string) => {
                        switch (value) {
                            case FilteringDate.Dt_kanban_column:
                               return currentValue.dt_kanban_column;
                            case FilteringDate.Updated_at:
                               return currentValue.updated_at;
                            case FilteringDate.Event_time:
                               return currentValue.event_time;
                            default:
                               return currentValue.event_time;
                        }
                    };
                    const createdDealDate = moment(
                       getDate(filterField) as string,
                       DEFAULT_DATE,
                    ).format(DEFAULT_DATE);
                    if (!acc[createdDealDate]) {
                        acc[createdDealDate] = [];
                    }
                    acc[createdDealDate].push(currentValue);
                }
                return acc;
            },
            {},
        );
